Arduino Intermedio

Sistema de Riego Automático

Controla el riego de tus plantas basado en la humedad del suelo.

  • Arduino Uno
  • Sensor de humedad del suelo
  • Módulo relé
  • Bomba de agua pequeña
  • Fuente de alimentación 12V 
// Código para sistema de riego automático
const int sensorPin = A0;
const int relayPin = 7;
int moistureLevel = 0;

void setup() {
  pinMode(relayPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  moistureLevel = analogRead(sensorPin);
  Serial.print("Humedad: ");
  Serial.println(moistureLevel);
  
  if (moistureLevel > 700) {  // Si el suelo está seco
    digitalWrite(relayPin, HIGH);  // Activa la bomba
    delay(2000);  // Riega por 2 segundos
    digitalWrite(relayPin, LOW);
  }
  delay(1000);  // Espera 1 segundo entre lecturas
}

Estación Meteorológica IoT

Mide temperatura, humedad y presión atmosférica y envíalos a la nube.

  • Sensor BME280
  • Pantalla OLED 0.96"

Adafruit_SSD1306 display(128, 64, &Wire, -1);

const char* ssid = "tu_SSID";
const char* password = "tu_PASSWORD";
const char* server = "api.thingspeak.com";
String apiKey = "TU_API_KEY";

void setup() {
  Serial.begin(115200);
  
  if(!bme.begin(0x76)) {
    Serial.println("Error con el sensor BME280");
    while(1);
  }
  
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  display.clearDisplay();
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
}

void loop() {
  float temp = bme.readTemperature();
  float hum = bme.readHumidity();
  float pres = bme.readPressure() / 100.0F;
  
  // Mostrar en pantalla OLED
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);
  display.setCursor(0,0);
  display.print("Temp: "); display.print(temp); display.println(" C");
  display.print("Hum: "); display.print(hum); display.println(" %");
  display.print("Pres: "); display.print(pres); display.println(" hPa");
  display.display();
  
  // Enviar a ThingSpeak
  if(WiFi.status()== WL_CONNECTED){
    HTTPClient http;
    String url = "http://" + String(server) + "/update?api_key=" + apiKey;
    url += "&field1=" + String(temp);
    url += "&field2=" + String(hum);
    url += "&field3=" + String(pres);
    
    http.begin(url);
    int httpCode = http.GET();
    http.end();
  }
  
  delay(30000); // Espera 30 segundos
}

Arduino Intermedio

Robot Seguidor de Línea

Un robot autónomo que sigue una línea negra sobre superficie clara.

  • Arduino Uno
  • Chasis de robot
  • Puente H L298N
  • Sensores TCRT5000
  • Batería 9V 
// Código para robot seguidor de línea
#define ENA 5
#define ENB 6
#define IN1 7
#define IN2 8
#define IN3 9
#define IN4 11

#define LS 10  // Sensor izquierdo
#define CS 4   // Sensor central
#define RS 2   // Sensor derecho

void setup() {
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  
  pinMode(LS, INPUT);
  pinMode(CS, INPUT);
  pinMode(RS, INPUT);
  
  analogWrite(ENA, 150);  // Velocidad motor A
  analogWrite(ENB, 150);  // Velocidad motor B
}

void loop() {
  int leftSensor = digitalRead(LS);
  int centerSensor = digitalRead(CS);
  int rightSensor = digitalRead(RS);
  
  // Línea detectada en sensor central
  if (centerSensor == LOW) {
    forward();
  } 
  // Línea detectada en sensor izquierdo
  else if (leftSensor == LOW && rightSensor == HIGH) {
    left();
  } 
  // Línea detectada en sensor derecho
  else if (rightSensor == LOW && leftSensor == HIGH) {
    right();
  } 
  // No se detecta línea
  else {
    stop();
  }
}

void forward() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void left() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void right() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
}

void stop() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}
